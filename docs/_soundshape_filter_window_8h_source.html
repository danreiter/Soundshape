265  void showAudioSettingsDialog()
266  {
267  DialogWindow::LaunchOptions o;
268 
269  int minNumInputs = std::numeric_limits<int>::max(), maxNumInputs = 0,
270  minNumOutputs = std::numeric_limits<int>::max(), maxNumOutputs = 0;
271 
272  auto updateMinAndMax = [](int newValue, int& minValue, int& maxValue)
273  {
274  minValue = jmin(minValue, newValue);
275  maxValue = jmax(maxValue, newValue);
276  };
277 
278  if (channelConfiguration.size() > 0)
279  {
280  auto defaultConfig = channelConfiguration.getReference(0);
281  updateMinAndMax((int)defaultConfig.numIns, minNumInputs, maxNumInputs);
282  updateMinAndMax((int)defaultConfig.numOuts, minNumOutputs, maxNumOutputs);
283  }
284 
285  if (auto* bus = processor->getBus(true, 0))
286  updateMinAndMax(bus->getDefaultLayout().size(), minNumInputs, maxNumInputs);
287 
288  if (auto* bus = processor->getBus(false, 0))
289  updateMinAndMax(bus->getDefaultLayout().size(), minNumOutputs, maxNumOutputs);
290 
291  minNumInputs = jmin(minNumInputs, maxNumInputs);
292  minNumOutputs = jmin(minNumOutputs, maxNumOutputs);
293 
294  o.content.setOwned(new SettingsComponent(*this, deviceManager,
295  minNumInputs,
296  maxNumInputs,
297  minNumOutputs,
298  maxNumOutputs));
299  o.content->setSize(500, 550);
300 
301  o.dialogTitle = TRANS("Audio/MIDI Settings");
302  o.dialogBackgroundColour = o.content->getLookAndFeel().findColour(ResizableWindow::backgroundColourId);
303  o.escapeKeyTriggersCloseButton = true;
304  o.useNativeTitleBar = true;
305  o.resizable = false;
306 
307  o.launchAsync();

564  if (!newMidiDevices.contains(oldDevice))
